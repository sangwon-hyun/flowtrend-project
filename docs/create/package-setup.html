Creating the flowtrend R package

3 Package setup

The DESCRIPTION file is created using this code.

usethis::create_package(
  path = ".",
  fields = list(
    Package = params$package_name,
    Version = "0.0.0.9000",
    Title = "flowtrend",
    Description = "Time-smooth mixture modeling for flow cytometry data (Hyun et al. 2025).",
    `Authors@R` = c(
        person(
            given = "Sangwon",
            family = "Hyun",
            email = "sangwonh@ucsc.edu",
            role = c("aut", "cre")),
        person(
            given = "Tim",
            family = "Coleman",
            email = "tscoleman226@gmail.com",
            role = c("aut")
        ),
        person(
            given = "Francois",
            family = "Ribalet",
            email = "ribalet@uw.edu",
            role = c("aut")
        ),
        person(
            given = "Jacob",
            family = "Bien",
            email = "jbien@usc.edu",
            role = c("aut")
        )
    )
  )
)
usethis::use_mit_license(copyright_holder = "Sangwon Hyun")
